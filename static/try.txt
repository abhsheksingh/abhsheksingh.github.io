let
    // Load all sheets from the Excel file
    Source = Excel.Workbook(File.Contents("C:\Path\To\Your\File.xlsx"), true),

    // Keep only the sheets you want
    FilteredSheets = Table.SelectRows(Source, each List.Contains({"Analysis", "Systems", "Topics", "Soft Skills"}, [Name])),

    // Process each sheet
    AddCustom = Table.AddColumn(FilteredSheets, "Processed", each
        let
            SheetData = [Data],

            // Get first 3 rows as headers
            First3Rows = Table.FirstN(SheetData, 3),

            // Transpose and fill down to combine headers
            Transposed = Table.Transpose(First3Rows),
            Filled = Table.FillDown(Transposed, Table.ColumnNames(Transposed)),

            // Combine into a single header row
            CombinedHeaders = Table.CombineColumns(Filled, Table.ColumnNames(Filled), each Text.Combine(List.RemoveNulls(_), "_"), "MergedHeaders"),

            // Promote the combined headers
            PromoteHeaders = Table.PromoteHeaders(CombinedHeaders, [PromoteAllScalars=true]),

            // Remove the first 3 rows from original data
            RemovedTopRows = Table.Skip(SheetData, 3),

            // Apply the new headers to the remaining data
            FinalTable = Table.RenameColumns(RemovedTopRows, List.Zip({Table.ColumnNames(RemovedTopRows), Table.ColumnNames(PromoteHeaders)})),

            // Add a Source_Sheet column
            AddSourceColumn = Table.AddColumn(FinalTable, "Source_Sheet", each [Name])
        in
            AddSourceColumn
    ),

    // Combine all processed sheets into one table
    Combined = Table.Combine(AddCustom[Processed])
in
    Combined
