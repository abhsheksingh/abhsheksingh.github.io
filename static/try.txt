let
    // Load all sheets from the Excel file
    Source = Excel.Workbook(File.Contents("C:\Path\To\Your\File.xlsx"), true),

    // Filter only the required sheets
    FilteredSheets = Table.SelectRows(Source, each List.Contains({"Analysis", "Systems", "Topics", "Soft Skills"}, [Name])),

    // Process each sheet
    AddCustom = Table.AddColumn(FilteredSheets, "Processed", each
        let
            SheetData = [Data],

            // Get first 3 rows
            First3Rows = Table.FirstN(SheetData, 3),

            // Combine first 3 rows into one header list
            HeaderList = List.Transform(
                Table.ToColumns(First3Rows),
                each Text.Combine(List.RemoveNulls(List.Transform(_, each Text.From(_))), "_")
            ),

            // Remove first 3 rows from original data
            RemovedTopRows = Table.Skip(SheetData, 3),

            // Apply new headers to remaining data
            AppliedHeaders = Table.PromoteHeaders(Table.FromColumns(Table.ToColumns(RemovedTopRows), HeaderList), [PromoteAllScalars=true]),

            // Add source sheet name
            AddSource = Table.AddColumn(AppliedHeaders, "Source_Sheet", each [Name])
        in
            AddSource
    ),

    // Combine all processed sheets
    Combined = Table.Combine(AddCustom[Processed])
in
    Combined
